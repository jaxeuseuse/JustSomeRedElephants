    function getMagic8BallAnswer() {
      const questionInput = document.getElementById('question');
      const answerElement = document.getElementById('magic8ball-answer');

      const question = questionInput.value.trim();

      if (question !== '') {
        fetch(`https://yesno.wtf/api?question=${question}`)
          .then(response => response.json())
          .then(data => {
            const answer = data.answer.charAt(0).toUpperCase() + data.answer.slice(1);
            answerElement.textContent = `Magic 8 Ball says: ${answer}`;
          })
          .catch(error => {
            console.error('Error fetching data:', error);
          });
      } else {
        answerElement.textContent = 'Please enter a question.';
      }
    }
  </script>
